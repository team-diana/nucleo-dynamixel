14 /* Class: SerialHalfDuplex
15  * A serial port (UART) for communication with other devices using
16  * Half-Duplex, allowing transmit and receive on a single
17  * shared transmit and receive line. Only one end should be transmitting
18  * at a time.
19  *
20  * Both the tx and rx pin should be defined, and wired together.
21  * This is in addition to them being wired to the other serial
22  * device to allow both read and write functions to operate.
23  *
24  * Example:
25  * > // Send a byte to a second HalfDuplex device, and read the response
26  * >
27  * > #include "mbed.h"
28  * >
29  * > // p9 and p10 should be wired together to form "a"
30  * > // p28 and p27 should be wired together to form "b"
31  * > // p9/p10 should be wired to p28/p27 as the Half Duplex connection
32  * >
33  * > SerialHalfDuplex a(p9, p10);
34  * > SerialHalfDuplex b(p28, p27);
35  * >
36  * > void b_rx() { // second device response
37  * > b.putc(b.getc() + 4);
38  * > }
39  * >
40  * > int main() {
41  * > b.attach(&b_rx);
42  * > for(int c = 'A'; c < 'Z'; c++) {
43  * > a.putc(c);
44  * > printf("sent [%c]\n", c);
45  * > wait(0.5); // b should respond
46  * > if(a.readable()) {
47  * > printf("received [%c]\n", a.getc());
48  * > }
49  * > }
50  * > }
51  *
52  * For Simplex and Full-Duplex Serial communication, see <Serial>
53  */

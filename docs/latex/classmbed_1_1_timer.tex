\subsection{Detailed Description}
A general purpose timer

Example\+: 
\begin{DoxyCode}
\textcolor{comment}{// Count the time to toggle a LED}

\textcolor{preprocessor}{#include "mbed.h"}

Timer timer;
DigitalOut led(LED1);
\textcolor{keywordtype}{int} begin, end;

\textcolor{keywordtype}{int} main() \{
    timer.start();
    begin = timer.read\_us();
    led = !led;
    end = timer.read\_us();
    printf(\textcolor{stringliteral}{"Toggle the led takes %d us"}, end - begin);
\}
\end{DoxyCode}
